int maxArea(int* height, int heightSize){
 int ans=0;
        for(int i=0;i<heightSize;i++)
                for(int j=i+1;j<heightSize;j++)
                {   if(((height[i]<height[j])? height[i]:height[j])*(j-i)>ans)
                        ans=((height[i]<height[j])? height[i]:height[j])*(j-i);}
        return ans;

}
